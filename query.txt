
CREATE TABLE ATTENDANCE_TB (
    RECORDID VARCHAR2(12 BYTE) PRIMARY KEY,
    PERSONNAME VARCHAR2(50 BYTE) NOT NULL,
    ATTENDANCE_DATE DATE NOT NULL,
    STATUS VARCHAR2(10 BYTE) NOT NULL,
    REMARKS VARCHAR2(100 BYTE),
    CONSTRAINT STATUS_CHECK CHECK (STATUS IN ('Present','Absent'))
);


CREATE SEQUENCE ATTENDANCE_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99
NOCYCLE;
